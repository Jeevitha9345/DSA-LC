class Solution {
    public int missingNumber(int[] nums) {
        int n=nums.length,i=0;
        while(i<n){
            int ind=nums[i];
            if(nums[i]<n && nums[i]!=nums[ind]){
                int temp=nums[i];
                nums[i]=nums[ind];
                nums[ind]=temp;
            }else{
                i++;
            }
        }for(i=0;i<n;i++){
            if(nums[i]!=i){
                return i;
            }
        }return n;
    }
}
