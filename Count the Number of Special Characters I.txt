class Solution {
    public int numberOfSpecialChars(String word) {
        HashMap<Character,Integer> hm=new HashMap<>();
        for(char c:word.toCharArray()){
            hm.put(c,hm.getOrDefault(c,0)+1);
        }
        int count=0;
        for(char i='a';i<='z';i++){
            if(hm.containsKey(i) && hm.containsKey(Character.toUpperCase(i))){
                count++;
            }
        }return count;
    }
}
