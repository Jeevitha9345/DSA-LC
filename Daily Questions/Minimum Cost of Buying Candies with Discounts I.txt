class Solution {
    public int minimumCost(int[] cost) {
        Arrays.sort(cost);
        int n=cost.length;
        int min=cost[0];
        int pay=0;
        if(n==1){
            return cost[0];
        }
        if(n==2){
            return cost[0]+cost[1];
        }
        for(int i=n-1;i>=0;i-=3){
            pay+=cost[i];
            if(i-1>=0){
                pay+=cost[i-1];
            }
        }return pay;
    }
}
