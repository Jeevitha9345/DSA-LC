class Solution {
    public int digitFrequencyScore(int n) {
        HashMap<Integer,Integer>hm=new HashMap<>();
        while(n>0){
            int rem=n%10;
            hm.put(rem,hm.getOrDefault(rem,0)+1);
            n/=10;
        }
        int score=0;
        for(Map.Entry<Integer,Integer>ent:hm.entrySet()){
            score+=ent.getKey()*ent.getValue();
        }return score;
    }
}
